<?xml version="1.0" encoding="utf-8"?>
<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
               xmlns:s="library://ns.adobe.com/flex/spark" 
               xmlns:mx="library://ns.adobe.com/flex/mx"
               creationComplete="creationCompleteHandler(event)" 
               resize="scaleDiagram()" 
               backgroundColor="#000000" 
               preloaderChromeColor="#000000" 
               viewSourceURL="srcview/index.html">
    <fx:Declarations>
        <!-- Place non-visual elements (e.g., services, value objects) here -->
    </fx:Declarations>
    <fx:Script>
        <![CDATA[
            import mx.events.FlexEvent;
            import mx.events.ResizeEvent;
            
            import org.robotlegs.RobotlegsDiagram;
            
            protected var diagram:RobotlegsDiagram;

            protected function creationCompleteHandler(event:FlexEvent):void
            {
                diagram = new RobotlegsDiagram();
                container.addChild(diagram);
                scaleDiagram();
            }

            protected function scaleDiagram():void
            {
                var scaleFactor:Number;
                
                scaleFactor = (height / 1114) * .9;
                
                if(diagram)
                {
                    diagram.scaleX = diagram.scaleY = scaleFactor;
                    if(diagram.width > width)
                        diagram.scaleX = diagram.scaleY = (width / 1616) * .9;
                    diagram.x = (width - diagram.width) / 2;
                    diagram.y = (height - diagram.height) / 2;
                }
            }

        ]]>
    </fx:Script>
    <mx:UIComponent id="container" width="{width}" height="{height}" />
</s:Application>
